  2. simple usage 
            
          function printMsg(...rest) {
            return function (e) {
              rest && rest.forEach(item => console.log(item));
              console.log(e.eventPhase);
              console.log(e.currentTarget);
            }
          }

          [
            'a',
            'comman+a',
          ].map(key => {
            keyMouseEvent.addDOMEvents().addKeyMouseEvent(key, printMsg(key));
          });

          [
            'b',
            'comman+b',
          ].map(key => {
            keyMouseEvent.addDOMEvents().addKeyMouseEvent(key, printMsg(key), { capture: true });
          });
